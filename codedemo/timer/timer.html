var test1 = new Timer({
    time: Date.now(),
    onStart: function(e){
        var s = e.year + '年' + e.month + '月' + e.date + '日 ' + e.hour + ':' + e.minute + ':' + e.second;
        $('#test1').html(s);
    },
    onChange: function(e){
        var s = e.year + '年' + e.month + '月' + e.date + '日 ' + e.hour + ':' + e.minute + ':' + e.second;
        $('#test1').html(s);
    }
});
